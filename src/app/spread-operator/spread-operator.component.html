üíª State Management with Spread:
// Adding items immutably
addTodo(): void {
  const newTodo = { id: this.todos.length + 1, text: 'New Task', completed: false };
  this.todos = [...this.todos, newTodo];
}

// Updating items immutably
toggleTodo(id: number): void {
  this.todos = this.todos.map(todo => 
    todo.id === id ? { ...todo, completed: !todo.completed } : todo
  );
}

// Removing items immutably
removeTodo(id: number): void {
  this.todos = this.todos.filter(todo => todo.id !== id);
}

‚úÖ Do's
  • Use spread for immutable updates
  • Copy arrays before modifications
  • Merge objects with spread syntax
  • Use with function arguments
  • Combine with destructuring
  • Use for state management
‚ùå Don'ts
  • Don't use for deep cloning
  • Avoid with circular references
  • Don't overuse in performance-critical code
  • Avoid with large objects unnecessarily
  • Don't rely on spread for complex state
  • Don't forget type safety
